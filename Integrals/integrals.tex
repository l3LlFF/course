{\bf Uniformization procedure.}

Denote  $h=\max\limits_{i=\overline{0,W}}(-D_0)_{ii}$.

Then the matrix  $P(n,t),\; n\ge 1,$ can be represented in the following form:
$$
P(n,t)=\sum_{j=0}^{\infty}e^{-h t}{(h
t)^j\over j!}K_n^{(j)},\; n\ge 0,  \eqno(1)
$$

where the matrices  $K_n^{(j)},\; n\ge 1, j\ge 0,$ are calculated by recursion

$$
K_0^{(0)}=I,\ K_n^{(0)}=O,\ n\ge 1,
$$ $$
K_0^{(j+1)}=K_0^{(j)}(I+h^{-1}D_0),
$$ $$
K_n^{(j+1)}=h^{-1}K_{n-1}^{(j)}D_1+K_n^{(j)}(I+h^{-1}D_0),\; n\ge 1,\; j\ge 0.
$$

{\bf Computation of transition probabilities }

We consider the constant service time. In this case, the distribution function $B(t)$ has the form
%\end{document}
$$
B(t)=\left\{
\begin{array}{cc}
0, & \;t\leq b_1,\\
1, & \;t>b_1.
\end{array}
\right.
$$






\begin{itemize}
\item[$\bullet$]



$\varphi_{k}(t) = \frac{(\gamma t)^{k} }{k!} e^{-\gamma t},\; k \geq 0$.

\item[$\bullet$]

$ \hat\varphi_{k}(t) = \sum\limits_{i=k}^{\infty} \varphi_{i}(t),\; k \geq 0$.

\item[$\bullet$]

$ \varphi_{k} =  \varphi_{k}(b_1),\; k \geq 0.$

\item[$\bullet$]
$ \hat\varphi_{k} = \sum\limits_{i=k}^{\infty} \varphi_{i},\; k \geq 0$.

\item[$\bullet$]
$
\sum\limits_{i=0}^{\infty} P(i,t) z^i = e^{(D_0+D_1z)t},\;|z|<1.$

\item[$\bullet$]

$ \Phi(i,k) =  \int\limits_{0}^{\infty} P(i,t)\varphi_{k}(t)dB(t)= P(i,b_1)\varphi_{k}(b_1)=e^{-(h+\gamma)b_1}\frac{(\gamma b_1)^k}{k!}\sum\limits_{j=0}^{\infty}\frac{(hb_1)^j}{j!}K_i^{(j)},\;$ $ i \geq 0,\; k \geq 0$.

\item[$\bullet$] 

$\hat \Phi(i,k) =  \int\limits_{0}^{\infty} P(i,t)\hat \varphi_{k}(t)dB(t)= P(i,b_1)\hat \varphi_{k}(b_1)=e^{-(h+\gamma)b_1}\sum\limits_{j=0}^{\infty}\frac{(hb_1)^j}{j!}K_i^{(j)}\sum\limits_{l=k}^{\infty}\frac{(\gamma b_1)^l}{l!},\;$ $ i \geq 0,\; k \geq 0$.

\item[$\bullet$]

$ N(m) = \int\limits_{0}^{\infty} P(m, t)\gamma e^{-\gamma t}dt=\gamma\sum\limits_{j=0}^{\infty}\frac{h^j}{j!}K_m^{(j)} \int\limits_{0}^{\infty}e^{-(h+\gamma) t}t^jdt=\frac{\gamma}{h+\gamma} \sum\limits_{j=0}^{\infty}(\frac{h}{h+\gamma})^jK_m^{(j)},\;$ $m \ge 0$.

\item[$\bullet$]

$ M(r) = \int\limits_{0}^{\infty} e^{D_{0}t}\varphi_{r}(t)D_{1}dt =
	\int\limits_{0}^{\infty} e^{D_{0}t}\frac{(\gamma t)^{r} }{r!} e^{-\gamma I t}D_{1}dt = \gamma^{r}(-D_{0} +  \gamma I)^{-(r+1)}D_{1}
$.

\item[$\bullet$]

$ \hat M(r) = \sum\limits_{l=r}^{\infty} M(l)=\gamma^{r}(-D_{0} +  \gamma I)^{-r}(-D_0)^{-1}D_{1},\; r \ge 0.$

\end{itemize}
